[V.Prestazione.gen_disj_complete]	
	ALL prest    Prestazione(prest) --> [
					// disjoint
					not EXISTS r, pe   
							r_isa_p(r, prest) and pe_isa_p(pe, prest)
					and
					// complete
					(  EXISTS x r_isa_p(x, prest) or pe_isa_p(x, prest) )
	
	].
